 68 /**
 69 * Ubica los componentes en la view dependiendo del presupuesto y uso del usuario
 70 * @method ubicarComponentes
 71 * @param {number} presupuesto - presupuesto del usuario
 72 * @param {string} uso - uso que el usuario le va a dar a la PC
 73 */
 74 function ubicarComponentes(presupuesto, uso) {
 75     let porcentajes = usos[uso];
 76     let p_cpu = porcentajes[0];
 77     let p_gpu = porcentajes[1];
 78     let p_ram = porcentajes[2];
 79     let p_ssd = porcentajes[3];
 80 
 81     document.getElementById("cpu").innerHTML = resultados.cpu; // default
 82     document.getElementById("gpu").innerHTML = resultados.gpu;
 83     document.getElementById("ram").innerHTML = resultados.ram;
 84     document.getElementById("ssd").innerHTML = resultados.ssd;
 85 
 86     for (const key of Object.keys(componentes.cpus)) {
 87         if (p_cpu * presupuesto >= componentes.cpus[key].precio) {
 88             resultados.cpu = key;
 89 
 90             document.getElementById("cpu").innerHTML = key;
 91         }
 92     }
 93 
 94     for (const key of Object.keys(componentes.gpus)) {
 95         if (p_gpu * presupuesto >= componentes.gpus[key].precio) {
 96             resultados.gpu = key;
 97 
 98             document.getElementById("gpu").innerHTML = key;
 99         }
100     }
101 
102     for (const key of Object.keys(componentes.rams)) {
103         if (p_ram * presupuesto >= componentes.rams[key].precio) {
104             resultados.ram = key;
105 
106             document.getElementById("ram").innerHTML = key;
107         }
108     }
109 
110     for (const key of Object.keys(componentes.ssds)) {
111         if (p_ssd * presupuesto >= componentes.ssds[key].precio) {
112             resultados.ssd = key;
113 
114             document.getElementById("ssd").innerHTML = key;
115         }
116     }
117 }

161  * @param {number} indice Indice de la celda donde ubicarla

163 function ubicar(componente, key, indice) {
164     var seleccionado = document.getElementById(componente + "-" + indice.toString());
165 
166     var title = seleccionado.children[0];
167     var opcion = title.children[0];

171     var rendimiento = more.children[1];

213 /**
214  * Dibuja el grafico en el canvas
215  * @method dibujar
216  */
217 function dibujar() {
218     
219     const cpu = document.getElementById('cpu').innerHTML;
220     const gpu = document.getElementById('gpu').innerHTML;
221     const ram = document.getElementById('ram').innerHTML;
222     const ssd = document.getElementById('ssd').innerHTML;
223     const total = componentes.cpus[cpu].precio + componentes.gpus[gpu].precio + componentes.rams[ram].precio +componentes.ssds[ssd].precio
224 
225     const cpu_porcentaje = componentes.cpus[cpu].precio / presupuesto;
226     const gpu_porcentaje = componentes.gpus[gpu].precio / presupuesto;
227     const ram_porcentaje = componentes.rams[ram].precio / presupuesto;
228     const ssd_porcentaje = componentes.ssds[ssd].precio / presupuesto;
229 
230     const canvas = document.getElementById('grafico');
231     const ctx = canvas.getContext('2d');
232     ctx.clearRect(0, 0, canvas.width, canvas.height);
233     ctx.font = "10px Arial";
234     ctx.fillStyle = "#000000";
235     ctx.fillText("Total gastado: $" + total.toString(), 10, 10);
236     const max_heigth =  canvas.height * 0.8;
237     const width = canvas.width;
238     const offset = 30;
239     const interval = (width - 2*offset)/4
240 
241     const cpu_h = max_heigth * cpu_porcentaje;
242     const gpu_h = max_heigth * gpu_porcentaje;
243     const ram_h = max_heigth * ram_porcentaje;
244     const ssd_h = max_heigth * ssd_porcentaje;
245 
246     ctx.fillStyle = "#2196f3";
247     ctx.fillRect(offset, canvas.height-cpu_h, 20, cpu_h);
248     ctx.fillText(Math.round(cpu_porcentaje * 100), offset - 20, canvas.height-5);
249     
250     ctx.fillRect(offset + interval + 10, canvas.height-gpu_h, 20, gpu_h);
251     ctx.fillText(Math.round(gpu_porcentaje * 100), offset + interval - 10, canvas.height-5);
252 
253     ctx.fillRect(offset + (interval + 10) * 2, canvas.height-ram_h, 20, ram_h);
254     ctx.fillText(Math.round(ram_porcentaje * 100), offset + interval*2, canvas.height-5);
255 
256     ctx.fillRect(offset + (interval + 10) * 3, canvas.height-ssd_h, 20, ssd_h);  
257     ctx.fillText(Math.round(ssd_porcentaje * 100), offset + interval*3 + 10, canvas.height-5);
258 }
