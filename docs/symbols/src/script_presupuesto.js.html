 67 /**
 68 * Ubica los componentes en la view dependiendo del presupuesto y uso del usuario
 69 * @method ubicarComponentes
 70 * @param {number} presupuesto - presupuesto del usuario
 71 * @param {string} uso - uso que el usuario le va a dar a la PC
 72 */
 73 function ubicarComponentes(presupuesto, uso) {
 74     let porcentajes = usos[uso];
 75     let p_cpu = porcentajes[0];
 76     let p_gpu = porcentajes[1];
 77     let p_ram = porcentajes[2];
 78     let p_ssd = porcentajes[3];
 79 
 80     document.getElementById("cpu").innerHTML = resultados.cpu; // default
 81     document.getElementById("gpu").innerHTML = resultados.gpu;
 82     document.getElementById("ram").innerHTML = resultados.ram;
 83     document.getElementById("ssd").innerHTML = resultados.ssd;
 84 
 85     for (const key of Object.keys(componentes.cpus)) {
 86         if (p_cpu * presupuesto <= componentes.cpus[key].precio) {
 87             resultados.cpu = key;
 88 
 89             document.getElementById("cpu").innerHTML = key;
 90             break;
 91         }
 92     }
 93 
 94     for (const key of Object.keys(componentes.gpus)) {
 95         if (p_gpu * presupuesto <= componentes.gpus[key].precio) {
 96             resultados.gpu = key;
 97 
 98             document.getElementById("gpu").innerHTML = key;
 99             break;
100         }
101     }
102 
103     for (const key of Object.keys(componentes.rams)) {
104         if (p_ram * presupuesto <= componentes.rams[key].precio) {
105             resultados.ram = key;
106 
107             document.getElementById("ram").innerHTML = key;
108             break;
109         }
110     }
111 
112     for (const key of Object.keys(componentes.ssds)) {
113         if (p_ssd * presupuesto <= componentes.ssds[key].precio) {
114             resultados.ssd = key;
115 
116             document.getElementById("ssd").innerHTML = key;
117             break;
118         }
119     }
120 }

160 /**
161  * Ubica el componente en su celda
162  * @param {string} componente Tipo de componente (cpu, gpu, ram o ssd)
163  * @param {string} key Llave del componente a ubicar
164  * @param {number} indice Indice de la celda donde ubicarla
165  */
166 function ubicar(componente, key, indice) {
167     var seleccionado = document.getElementById(componente + "-" + indice.toString());
168 
169     var title = seleccionado.children[0];
170     var opcion = title.children[0];
171 
172     var more = seleccionado.children[1];
173     var precio = more.children[0];
174     var rendimiento = more.children[1];
175 
176     var object = componentes[componente + "s"][key];
177 
178     opcion.innerHTML = key;
179     precio.innerHTML = "Precio: " + object.precio;
180     rendimiento.innerHTML = "Rendmiento: " + object.rendimiento;
181 }

183 /**
184 * Cambia el componente actual por el seleccionado y calcula el nuevo grafico
185 * @method cambiar
186 * @param {string} componente - componente de la pc (cpu, gpu, ram o ssd)
187 * @param {string} indice - indice del boton clickeado
188 */
189 function cambiar(componente, indice) {
190     var curr = document.getElementById(componente);
191     var seleccionado = document.getElementById(componente + "-" + indice.toString());
192     var title = seleccionado.children[0];
193     var opcion = title.children[0];
194 
195     var key = curr.innerHTML;
196     curr.innerHTML = opcion.innerHTML;
197 
198     resultados[componente] = curr.innerHTML;
199 
200     ubicar(componente, key, indice);
201     dibujar();
202 }
