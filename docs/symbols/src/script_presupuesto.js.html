 59 /**
 60 * Carga la view con la informacion necesaria
 61 * @method loadPresupuesto
 62 */
 63 function loadPresupuesto() {
 64     let params = new URLSearchParams(location.search);
 65     let usuario = params.get("usuario");
 66     let presupuesto = params.get("presupuesto");
 67     let uso = params.get("uso");
 68 
 69     document.getElementById("header").innerHTML = usuario + " ya tenemos tu PC!"
 70 
 71     ubicarComponentes(presupuesto, uso);
 72 }

 74 /**
 75 * Ubica los componentes en la view dependiendo del presupuesto y uso del usuario
 76 * @method ubicarComponentes
 77 * @param {number} presupuesto - presupuesto del usuario
 78 * @param {string} uso - uso que el usuario le va a dar a la PC
 79 */
 80 function ubicarComponentes(presupuesto, uso) {
 81     let porcentajes = usos[uso];
 82     let p_cpu = porcentajes[0];
 83     let p_gpu = porcentajes[1];
 84     let p_ram = porcentajes[2];
 85     let p_ssd = porcentajes[3];
 86     let resultados = [tier3, tier3, "64gb", "2tb"];
 87 
 88     for (const key of Object.keys(cpus)) {
 89         if (p_cpu * presupuesto <= key) {
 90             resultados[0] = key;
 91 
 92             var cpu = Object.keys(cpus[key])[0]
 93             document.getElementById("cpu").innerHTML = cpu;
 94             break;
 95         }
 96     }
 97 
 98     for (const key of Object.keys(gpus)) {
 99         if (p_gpu * presupuesto <= key) {
100             resultados[1] = key;
101 
102             var gpu = Object.keys(gpus.key)[0]
103             document.getElementById("gpu").innerHTML = gpu;
104             break;
105         }
106     }
107 
108     for (const key of Object.keys(rams)) {
109         if (p_ram * presupuesto <= rams[key].precio) {
110             resultados[2] = key;
111 
112             var ram = Object.keys(rams[key])
113             document.getElementById("ram").innerHTML = ram;
114             break;
115         }
116     }
117 
118     for (const key of Object.keys(ssds)) {
119         if (p_ssd * presupuesto <= ssds[key].precio) {
120             resultados[3] = key;
121 
122             var ssd = Object.keys(ssds[key])
123             document.getElementById("ssd").innerHTML = ssd;
124             break;
125         }
126     }
127 }
